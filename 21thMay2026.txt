-- FOREIGN KEY WITH ON DELETE CASCADE

    dept_id INT,
    FOREIGN KEY (dept_id)

);

-- FACULTY TABLE QUERIES

SELECT *
FROM Faculty;

SELECT FIRST_NAME, DEPARTMENT, SALARY
FROM Faculty
WHERE DEPARTMENT='ECE'
ORDER BY SALARY ASC;

SELECT FIRST_NAME, DEPARTMENT, SALARY
FROM Faculty
WHERE DEPARTMENT='ECE'
ORDER BY SALARY DESC;

-- GROUP BY
SELECT DEPARTMENT, COUNT(DEPARTMENT)
FROM Faculty
GROUP BY DEPARTMENT;

SELECT DEPARTMENT, COUNT(FACULTY_ID)
FROM Faculty
GROUP BY DEPARTMENT
ORDER BY DEPARTMENT;

SELECT DEPARTMENT,
       SUM(SALARY) AS TOTAL_SALARY
FROM Faculty
GROUP BY DEPARTMENT
ORDER BY TOTAL_SALARY DESC;

-- HAVING CLAUSE

SELECT DEPARTMENT,
       COUNT(FACULTY_ID)
FROM Faculty
GROUP BY DEPARTMENT
HAVING COUNT(FACULTY_ID) >= 2;

SELECT DEPARTMENT,
       SUM(SALARY) AS TOTAL_SALARY
FROM Faculty
GROUP BY DEPARTMENT
HAVING TOTAL_SALARY < 250000;

-- LIMIT & OFFSET

SELECT *
FROM Faculty
LIMIT 5 OFFSET 2;

-- SENIOR FACULTY TABLE

CREATE TABLE SeniorFaculty(
    FACULTY_ID INT NOT NULL PRIMARY KEY,
    FIRST_NAME CHAR(25),
    LAST_NAME CHAR(25),
    SALARY INT,
    JOINING_DATE DATETIME,
    DEPARTMENT CHAR(25)
);

-- SUBQUERY

SELECT FACULTY_ID,
       FIRST_NAME,
       DEPARTMENT
FROM Faculty
WHERE FACULTY_ID IN
(
    SELECT FACULTY_ID
    FROM SeniorFaculty
);

-- LOWEST TWO DEPARTMENT SALARIES

SELECT DEPARTMENT,
       SUM(SALARY) AS TOTAL_SALARY
FROM Faculty
GROUP BY DEPARTMENT
ORDER BY TOTAL_SALARY ASC
LIMIT 2;

-- SECOND HIGHEST SALARY IN ECE

SELECT *
FROM Faculty
WHERE DEPARTMENT='ECE'
ORDER BY SALARY DESC
LIMIT 1 OFFSET 1;

-- SECOND HIGHEST SALARY IN EACH DEPARTMENT

SELECT FIRST_NAME,
       LAST_NAME,
       SALARY,
       DEPARTMENT
FROM Faculty F1
WHERE SALARY =
(
    SELECT SALARY
    FROM Faculty F2
    WHERE F1.DEPARTMENT = F2.DEPARTMENT
    ORDER BY SALARY DESC
    LIMIT 1 OFFSET 1
);
